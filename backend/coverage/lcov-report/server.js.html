       FROM TIERRA_EN_CALMA.USUARIOS
       WHERE LOWER(CORREO_ELECTRONICO) = LOWER(:correo_electronico)
       AND CONTRASENA = :contrasena`,

      const usuario = result.rows[0];

        user: usuario,
